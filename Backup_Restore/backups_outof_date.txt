

DECLARE @MaxDaysFull int = 8
DECLARE @MaxHrsDiff int = 49
DECLARE @MaxMinsLog int = 361
 
;with backup_info as
(
select
database_name,
type backup_type,
backup_finish_date,
rownum =
row_number() over
(
partition by database_name, type
order by backup_finish_date desc
)
from msdb.dbo.backupset bs WITH (NOLOCK)
)
,backup_part as
(
select
db.name as database_name,
type backup_type,
backup_finish_date,
rownum =
row_number() over
(
partition by database_name, type
order by backup_finish_date desc
)
from msdb.dbo.backupset bs right join sys.databases db on bs.database_name = db.name
)
,database_info as
(
select dbs.name database_name,
sdb.status,
sdb.crdate,
dbs.recovery_model,
1 as rownum from sys.databases dbs
join master..sysdatabases sdb on dbs.name=sdb.name
where (sdb.status & 512)=0
)
 
select distinct
a.database_name,
a.crdate,
a.recovery_model,
f.backup_finish_date full_date,
CASE WHEN f.backup_finish_date is NULL
THEN CASE WHEN a.database_name in ('msdb','master','model','tempdb','distribution') THEN 0
WHEN DATEDIFF(d,a.crdate,GETDATE()) < @MaxDaysFull THEN 0
ELSE 99999
END
ELSE DATEDIFF (DAY, f.backup_finish_date, GETDATE())
END AS Full_AgedDays,
d.backup_finish_date diff_date,
CASE WHEN d.backup_finish_date is NULL
THEN CASE WHEN a.database_name IN ('msdb','master','model','tempdb','distribution') THEN 0
WHEN DATEDIFF(d,a.crdate,GETDATE()) < @MaxDaysFull THEN 0
ELSE 99999
END
ELSE DATEDIFF (HOUR, d.backup_finish_date, GETDATE())
END AS Diff_AgedHrs,
l.backup_finish_date log_date,
CASE WHEN l.backup_finish_date is NULL
THEN CASE WHEN a.recovery_model != 1 THEN 0
WHEN a.database_name IN ('msdb','master','model','tempdb','distribution') THEN 0
WHEN DATEDIFF(d,a.crdate,GETDATE()) < @MaxDaysFull THEN 0
ELSE 9999999
END
ELSE CASE WHEN a.recovery_model != 1 THEN 0
ELSE DATEDIFF (MINUTE, l.backup_finish_date, GETDATE())
END
END AS Log_AgedMins
from database_info a LEFT OUTER JOIN backup_info b ON a.database_name = b.database_name and a.rownum = b.rownum
LEFT JOIN backup_info f ON a.database_name = f.database_name and a.rownum = f.rownum and f.rownum=1 and f.backup_type = 'D'
LEFT JOIN backup_info d ON a.database_name = d.database_name and a.rownum = d.rownum and d.rownum=1 and d.backup_type = 'I'
LEFT JOIN backup_info l ON a.database_name = l.database_name and a.rownum = l.rownum and l.rownum=1 and l.backup_type = 'L'
where a.rownum = 1
and ((a.status & 512)=0)
and a.database_name not in ('msdb','master','model','tempdb','distribution')
and a.database_name not in (SELECT name from master.sys.databases where ((status & 512)=0) and OBJECT_ID(a.database_name+'.dbo.MSreplication_objects') IS NOT NULL)




DECLARE @is_primary sysname;
    DECLARE @db_names TABLE (name varchar(100));
    DECLARE @GraceHrs int = 12
    DECLARE @MaxHrsFull int = 192
    DECLARE @MaxHrsDiff int = 24
    DECLARE @MaxMinsLog int = 361

    IF SERVERPROPERTY('IsHadrEnabled')= 1
    BEGIN
       SET @is_primary = (

SELECT
   
 RCS.replica_server_name -- SQL cluster node name

FROM
sys.availability_groups_cluster AS AGC
  INNER JOIN sys.dm_hadr_availability_replica_cluster_states AS RCS
   ON
    RCS.group_id = AGC.group_id
  INNER JOIN sys.dm_hadr_availability_replica_states AS ARS
   ON
    ARS.replica_id = RCS.replica_id
  INNER JOIN sys.availability_group_listeners AS AGL
   ON
    AGL.group_id = ARS.group_id
WHERE
ARS.role_desc = 'PRIMARY'
)


                    IF (@is_primary = @@SERVERNAME)
                    BEGIN
                           INSERT INTO @db_names 
                                  SELECT name FROM sys.databases WITH(NOLOCK)
                                  WHERE name <> 'tempdb'
                    END
                    ELSE
                    BEGIN
                           INSERT INTO @db_names
                                  SELECT name FROM sys.databases AS dbs WITH(NOLOCK)
                                  LEFT JOIN sys.availability_databases_cluster AS agdbs WITH(NOLOCK)
                                  ON dbs.name = agdbs.database_name
                                  WHERE agdbs.database_name IS NULL
                    END
    END
    ELSE
                            INSERT INTO @db_names 
                                  SELECT name FROM sys.databases WITH(NOLOCK)
                                  WHERE name <> 'tempdb'

;with backup_info as
(
    select
              database_name,
        type backup_type, 
        backup_finish_date,
        rownum = 
            row_number() over
            (
                partition by database_name, type 
                order by backup_finish_date desc
            )
    from msdb.dbo.backupset bs WITH (NOLOCK)
)
,backup_part as
(
    select
        db.name as database_name,
        type backup_type,
        backup_finish_date,
        rownum = 
            row_number() over
            (
                partition by database_name, type 
                order by backup_finish_date desc
            )
    from msdb.dbo.backupset bs WITH (NOLOCK) right join sys.databases db WITH (NOLOCK) on bs.database_name = db.name
       and DB_NAME(database_id) IN (SELECT name FROM @db_names) where db.state_desc='ONLINE'
)
,database_info as
(
select dbs.name database_name, 
              sdb.status, 
              sdb.crdate, 
        dbs.is_in_standby,
              dbs.recovery_model,
              1 as rownum from sys.databases dbs WITH (NOLOCK)
join master..sysdatabases sdb WITH (NOLOCK) on dbs.name=sdb.name
where (sdb.status & 512)=0 AND dbs.is_in_standby!=1 AND dbs.state_desc='ONLINE'
and DB_NAME(database_id) IN (SELECT name FROM @db_names)
)

SELECT 
              COALESCE(SUM(CASE WHEN Full_AgedHrs >= @MaxHrsFull THEN 1 ELSE 0 END),0) FullsOutOfDate,
        COALESCE(SUM(CASE WHEN Full_AgedHrs < @MaxHrsFull THEN 1 ELSE 0 END),0) FullsWithinDate,
              COALESCE(SUM(CASE WHEN Diff_AgedHrs >= @MaxHrsDiff THEN 1 ELSE 0 END),0) DiffsOutOfDate,
        COALESCE(SUM(CASE WHEN Diff_AgedHrs < @MaxHrsDiff THEN 1 ELSE 0 END),0) DiffsWithinDate,
              COALESCE(SUM(CASE WHEN Log_AgedMins >= @MaxMinsLog THEN 1 ELSE 0 END),0) LogsOutOfDate,
        COALESCE(SUM(CASE WHEN Log_AgedMins < @MaxMinsLog THEN 1 ELSE 0 END),0) LogsWithinDate,      
        COALESCE(SUM(1),0) TotalDatabaseCount
         
FROM (
select distinct
    a.database_name,
       a.crdate,
    a.recovery_model,
    f.backup_finish_date full_date,
    CASE WHEN f.backup_finish_date is NULL
         THEN CASE WHEN a.database_name in ('msdb','master','model','tempdb','distribution','SaaSAutomation') THEN 0
                   WHEN DATEDIFF(d,a.crdate,GETDATE()) < @MaxHrsFull THEN 0 -- database created since last full backup cycle - dont consider yet
              ELSE 99999
       END
       ELSE DATEDIFF  (HOUR, f.backup_finish_date, GETDATE())
       END AS Full_AgedHrs,
    d.backup_finish_date diff_date,
    CASE WHEN d.backup_finish_date is NULL
      THEN CASE WHEN a.database_name IN ('msdb','master','model','tempdb','distribution','SaaSAutomation') THEN 0
                WHEN DATEDIFF(d,a.crdate,GETDATE()) < @MaxHrsFull THEN 0 -- database created since last full backup cycle - dont consider yet
           ELSE 99999
      END
       ELSE DATEDIFF  (HOUR, d.backup_finish_date, GETDATE())
       END AS Diff_AgedHrs,
    l.backup_finish_date log_date,
    CASE WHEN l.backup_finish_date is NULL
      THEN CASE WHEN a.recovery_model != 1 THEN 0
                         WHEN a.database_name IN ('msdb','master','model','tempdb','distribution','SaaSAutomation') THEN 0    
                   WHEN DATEDIFF(d,a.crdate,GETDATE()) < @MaxHrsFull THEN 0 -- database created since last full backup cycle - dont consider yet
           ELSE 9999999
       END
       ELSE CASE WHEN a.recovery_model != 1 THEN 0
            ELSE DATEDIFF  (MINUTE, l.backup_finish_date, GETDATE())
            END
       END AS Log_AgedMins
from database_info a LEFT OUTER JOIN backup_info b ON a.database_name = b.database_name and a.rownum = b.rownum
       LEFT JOIN backup_info f ON a.database_name = f.database_name and a.rownum = f.rownum and f.rownum=1 and f.backup_type = 'D'
       LEFT JOIN backup_info d ON a.database_name = d.database_name and a.rownum = d.rownum and d.rownum=1 and d.backup_type = 'I'
       LEFT JOIN backup_info l ON a.database_name = l.database_name and a.rownum = l.rownum and l.rownum=1 and l.backup_type = 'L' 
where a.rownum = 1
and ((a.status & 512)=0) -- online
and a.database_name not in ('msdb','master','model','tempdb','distribution')
and a.database_name not in (SELECT name from master.sys.databases where ((status & 512)=0) AND state_desc='ONLINE' and OBJECT_ID(a.database_name+'.dbo.MSreplication_objects') IS NOT NULL) -- not subscriber db for replication
) as t

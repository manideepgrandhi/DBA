-- Long running queries – This will tell you all long running queries. And give you the command to delete them if necessary.
--
     SELECT sqltext.TEXT
          , req.session_id
          , req.status
          , req.command
          , req.start_time
          , req.cpu_time
          , req.total_elapsed_time as 'elapsed time (milliseconds)'
          , req.reads
          , req.writes
          , req.logical_reads
          , req.session_id
          , sess.login_time
          , sess.host_name
          , sess.program_name
          , sess.login_name
          , sess.last_request_start_time
          , 'Kill '+cast( req.session_id as varchar(10)) as 'Kill command'
       FROM sys.dm_exec_requests req
CROSS APPLY sys.dm_exec_sql_text(req.sql_handle) AS sqltext
          , sys.dm_exec_sessions as sess
where sess.session_id = req.session_id

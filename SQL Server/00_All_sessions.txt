
use CODBAProcedures
go
exec sp_WhoIsActive @get_additional_info=1,
@show_sleeping_spids=1,@get_task_info = 2 
go



SELECT
r.session_id
, r.start_time
, TotalElapsedTime_ms = r.total_elapsed_time
, r.[status]
, r.command
, DatabaseName = DB_Name(r.database_id)
, r.wait_type
, r.last_wait_type
, r.wait_resource
, r.cpu_time
, r.reads
, r.writes
, r.logical_reads
, t.[text] AS [executing batch]
, SUBSTRING(
t.[text], r.statement_start_offset / 2,
( CASE WHEN r.statement_end_offset = -1 THEN DATALENGTH (t.[text])
ELSE r.statement_end_offset
END - r.statement_start_offset ) / 2
) AS [executing statement]
, p.query_plan
,r.plan_handle
FROM
sys.dm_exec_requests r
CROSS APPLY
sys.dm_exec_sql_text(r.sql_handle) AS t
CROSS APPLY
sys.dm_exec_query_plan(r.plan_handle) AS p
where r.session_id<>@@SPID
ORDER BY
r.total_elapsed_time DESC;

SELECT s.[session_id] AS 'spid', s.[host_process_id] AS 'host_PID', db_name(r.[database_id]) as 'dbname',
  s.[login_name] as 'login_name', s.[host_name] AS 'host_name', r.[command],
  r.[status] as 'status', r.[total_elapsed_time]/1000 AS 'duration',
  'wait_type' =
   (CASE
    WHEN r.[wait_type] is null THEN 'NONE'
    ELSE r.[wait_type]
   END),
  r.[wait_time]/1000 AS 'wait_time', r.[logical_reads], r.[writes], r.[cpu_time],
  r.[blocking_session_id] as 'blocker_spid', s.[last_request_start_time], t.[text] as 'sql',
  r.[sql_handle]
FROM sys.dm_exec_sessions s
LEFT OUTER JOIN sys.dm_exec_requests r ON s.[session_id] = r.[session_id]
OUTER APPLY sys.dm_exec_sql_text(r.sql_handle) AS t
WHERE s.[login_name] <> 'sa' and s.[session_id] <> @@SPID
AND s.[status] = 'running'
--AND s.[session_id] = 121
--AND s.[login_name] = 'xxxxxx'
ORDER BY 'duration' desc
go

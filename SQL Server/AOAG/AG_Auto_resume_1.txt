declare @dbname nvarchar(512)
declare @TSQL NVARCHAR (512)
Declare db_cursor cursor local fast_forward
 for
SELECT agdb.database_name gdbname from sys.availability_databases_cluster agdb join sys.availability_groups_cluster ag
on agdb.group_id=ag.group_id join
sys.dm_hadr_database_replica_states agdbrs on agdb.group_database_id=agdbrs.group_database_id join sys.dm_hadr_database_replica_cluster_states agrcs on agdb.group_database_id =agrcs.group_database_id join
sys.availability_replicas ar on agrcs.replica_id=ar.replica_id
where ar.replica_server_name=@@servername
and Synchronization_health=0
open db_cursor
Fetch next from db_cursor into @dbname
While @@fetch_status=0

Begin
--then resume back
--print '--resuming data movement for database '  +@dbname;
Set @tsql =  'Alter Database ['  + @dbname + '] set hadr resume'

--uncomment following line for direct execution
--execute sp_executesql @tsql
Print @tsql
Fetch next from db_cursor into @dbname
End
Close db_cursor
Deallocate db_cursor
